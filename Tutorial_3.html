    6.9, 7.8, 8.9, 5.2, 7.7, 9.6, 8.7,

# Rainfall categories (in inches)
rainfall_categories = ['4 to <5', '5 to <6', '6 to <7', '7 to <8', '8 to <9', 
                       '9 to <10', '10 to <11', '11 to <12', '12 to <13', '13 to <14']

# Midpoints of each rainfall category for plotting purposes
midpoints = [4.5, 5.5, 6.5, 7.5, 8.5, 9.5, 10.5, 11.5, 12.5, 13.5]

# Relative frequencies
relative_frequencies = [0.052, 0.103, 0.086, 0.103, 0.172, 0.069, 0.207, 0.103, 0.052, 0.052]
In [6]:
# Calculate the cumulative relative frequency
cumulative_relative_frequencies = [sum(relative_frequencies[:i+1]) for i in range(len(relative_frequencies))]

# Create a dot plot for the cumulative relative frequency
plt.plot(midpoints, cumulative_relative_frequencies, 'o-')

plt.xlabel('Annual Rainfall (in inches)', fontsize=20)
plt.ylabel('Cumulative Relative Frequency', fontsize=20)

# Display the plot
plt.show()
